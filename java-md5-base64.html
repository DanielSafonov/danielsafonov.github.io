Java get file md5 and base64

18 Jul 2019 | java 
//byte[] to String Base64
public static String byteToBase64(byte[] data) {
	return Base64.getMimeEncoder().encodeToString(data);
}

//String Base64 to byte[]
public static byte[] base64ToByte(String base64){
	return Base64.getMimeDecoder().decode(base64);
}


//byte[] to String MD5
public static String getByteArrayFileHash(byte[] fileData) throws Exception {
	try {
		MessageDigest md = MessageDigest.getInstance("MD5");
		md.update(fileData);
		byte[] digest = md.digest();
		return DatatypeConverter.printHexBinary(digest).toUpperCase();
	} catch (Exception e) {
		throw new Exception("An error in getByteArrayFileHash() while calculating hash code!", e);
	}
}
