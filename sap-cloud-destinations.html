SAP Cloud Destinations

04 Jun 2019 | sap java

SAP Cloud Platform Connectivity provides two services for the Cloud Foundry environment, the Connectivity service and the Destination service. <..> Using the Destination service, you can retrieve and store the technical information about the target resource (destination) that you need to connect your application to a remote service or system.

Connectivity in the Cloud Foundry Environment

SAP Cloud > Java Application > Configuration > Destitations

  • Name - имя Destination, используется для получения его параметров.
  • Type - HTTP, LDAP, MAIL, RFC
  • Description - описание
  • URL - адрес
  • Proxy Type - Internet, OnPremise
  • Authentication - AppToAppSSO, BasicAuthentication, ClientCertificateAuthentication, NoAuthentication, OAuth2ClientCredentials, OAuth2SAMLBearerAssertion, SAPAssertionSSO
  • Additional Properties - дополнительные поля (пара “имя - значение”)

DestinationGetter.java

import com.sap.core.connectivity.api.configuration.ConnectivityConfiguration;
import com.sap.core.connectivity.api.configuration.DestinationConfiguration;
import lombok.AllArgsConstructor;
import lombok.Getter;
import lombok.extern.slf4j.Slf4j;

import javax.naming.InitialContext;

/**
 * Класс для получения данных из SAP Cloud Destinations
 *
 * @author Daniel Safonov
 */
@AllArgsConstructor
@Getter
@Slf4j
public class DestinationsGetter {
	private final String URL;
	private final String USER;
	private final String PASSWORD;
	private DestinationConfiguration DESTINATION;

	//Соединиться с выбранным Destination и получить его данные
	public static DestinationsGetter connectFrom(String destinationName) throws Exception {
		//Получить destination
		InitialContext context = new InitialContext();
		ConnectivityConfiguration configuration = (ConnectivityConfiguration) context.lookup("java:comp/env/connectivityConfiguration");
		DestinationConfiguration destination = configuration.getConfiguration(destinationName);

		if (destination == null) {
			log.error("Destination not found: " + destinationName);
			throw new Exception("Destination not found: " + destinationName);
		}

		//Получить параметры
		String user = destination.getProperty("User");
		String password = destination.getProperty("Password");
		String url = destination.getProperty("URL");

		if (StringUtils.isEmpty(user)) {
			log.error("Destination: User is null or empty");
			throw new Exception("Destination: User is null or empty");
		}
		if (StringUtils.isEmpty(password)) {
			log.error("Destination: Password is null or empty");
			throw new Exception("Destination: Password is null or empty");
		}
		if (StringUtils.isEmpty(url)) {
			log.error("Destination: Url is null or empty");
			throw new Exception("Destination: Url is null or empty");
		}

		log.info("Destination is correct! User: " + user + " URL: " + url);
		return new DestinationsGetter(user, password, url, destination);
	}

	//Получить переменную из окружения SAP Cloud по имени
	public String getProperty(String name) throws Exception {
		String value = DESTINATION.getProperty(name);
		if (value == null || value.isEmpty()) {
			log.error("Destination property not found:" + name);
			throw new Exception("Destination property not found: " + name);
		}
		return value;
	}
}
